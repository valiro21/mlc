    #include <iostream>
    #include <deque>
    using namespace std;
    void printKMax(int arr[], int n, int k){
        deque<pair<int, int > > dq;

        for (int i = 0; i < k; ++i){
            while (!dq.empty() && dq.back().first <= arr[i])
                dq.pop_back();
            dq.push_back({arr[i], i});
        }
        cout << dq.front().first << ' ';

        for (int i = k; i < n; ++i){
            while (!dq.empty() && dq.back().first <= arr[i])
                dq.pop_back();
            dq.push_back({arr[i], i});
            if (dq.front().second <= i - k) dq.pop_front();

            cout << dq.front().first << ' ';
        }
        cout << '\n';
    }
